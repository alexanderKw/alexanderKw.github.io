      /* function first */
      function calcSum1(a, b) {
        a = +prompt('Введите первое число', '0');
        b = +prompt('Введите второе число', '0');
        var sum = 0;
  
        if (!a || !b) {
          return 'Пользователь отменил ввод чисел';
        }
  
        if (!isNaN(a) && !isNaN(b)) {
          sum = a + b;
        } else {
          alert('Вводите только числа!');
          return calcSum1();
        }
  
        return sum;
      }

      /* function second */
      function calcSum2(nums) {
        nums = prompt('Введите два числа через запятую', '0, 0');
  
        if (!nums) {
          return 'Пользователь отменил ввод чисел';
        }
  
        var arrNums = nums.split(',');
        var sum = 0;
  
        for (var i = 0; i < arrNums.length; i++) {
          arrNums[i] = +arrNums[i];
          if (!isNaN(arrNums[i])) {
            sum += arrNums[i];
          } else {
            alert('Вводите только числа!');
            return calcSum2();
          }
        }
  
        return sum;
      }
