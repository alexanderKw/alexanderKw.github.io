      var randSimbol = [10.5, 15, 112, 77.6];

      function getFixedNums(arr) {
        for (var i = 0; i < randSimbol.length; i++) {
          if (!isNaN(arr[i])) {
            arr[i] = +arr[i];
            arr[i] = arr[i].toFixed(2);
          } else {
            arr[i] = '"' + arr[i] + '"' + ' - это не число';
          }
        }
  
        return arr;
      }
  
      document.write(getFixedNums(randSimbol));
