      var randomArray = new Array(10);

      function getRandom(arr, min, max) {
        for (var i = 0; i < arr.length; i++) {
          var randNum = Math.floor((Math.random() * (max - min + 1)) + min);
          arr[i] = randNum;
        }
  
        return arr;
      }
  
      document.write('[ ' + getRandom(randomArray, 1, 20) + ' ]');
