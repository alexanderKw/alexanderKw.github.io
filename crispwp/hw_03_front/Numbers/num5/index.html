      var numberArray = [34.82, 12.90, 17.01, 78.51];

      function numsRounding(arr, method) {
        for (var i = 0; i < arr.length; i++) {
          switch (method) {
            case 'round':
              arr[i] = Math.round(arr[i]);
              break;
            case 'ceil':
              arr[i] = Math.ceil(arr[i]);
              break;
            case 'floor':
              arr[i] = Math.floor(arr[i]);
              break;
          }
        }
  
        return arr;
      }
  
      document.write('[ ' + numsRounding(numberArray, 'floor') + ' ]');
