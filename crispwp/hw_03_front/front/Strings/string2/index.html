      var userWord = prompt('Введите любое слово', '');

      function getNumCharacters(string) {
        var str = string.trim().length;
        var result;
  
        if (str == 10) {
          result = 'В этом слове 10 символов';
        } else if (str > 10) {
          result = 'В этом слове больше 10-и символов';
        } else {
          result = 'В этом слове меньше 10-и символов';
        }
  
        return result;
      }
  
      document.write(getNumCharacters(userWord));
