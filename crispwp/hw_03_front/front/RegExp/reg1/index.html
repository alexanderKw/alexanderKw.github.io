      function getFirstSymbol(str) {
        var reg = new RegExp(/\d/);
        var result = str.search(reg);
        var answer = '';
  
        if (result == 0) {
          answer = 'Содержимое строки начинается с цифры';
        } else {
          answer = 'Содержимое строки не начинается с цифры';
        }
  
        document.write(answer);
      }
