      function isNumeric(arrElms) {
        for (var i = 0; i < arrElms.length; i++) {
          if (!isNaN(parseFloat(arrElms[i]) && isFinite(arrElms[i]))) {
            document.write(arrElms[i] + ' - Число' + '<br>');
          } else {
            document.write(arrElms[i] + ' - Не число' + '<br>');
          }
        }
      }
