Задание:

Есть объект salaries с зарплатами. Напишите код, который выведет сумму всех зарплат. Если объект пустой, то результат должен быть 0.
var salaries = {
"Вася": 100,
"Петя": 300,
"Даша": 250
};
//... 650

Код:

      var salaries = {
        'Вася': 100,
        'Петя': 300,
        'Даша': 250
      };
      
      function getSumSalaries(obj) {
        var sum = 0;
      
        for (var key in obj) {
          if (!isNaN(obj[key])) {
            sum += +obj[key];
          }
        }
      
        return sum;
      }
      
      document.write(getSumSalaries(salaries));
