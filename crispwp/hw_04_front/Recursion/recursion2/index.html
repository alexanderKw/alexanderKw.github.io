Задание:

Напишите функцию которая будет спрашивать у пользователя число(n) и будет вычислять сумму чисел от 1 до n, например:
sumTo(1) = 1
sumTo(2) = 2 + 1 = 3
sumTo(3) = 3 + 2 + 1 = 6
sumTo(4) = 4 + 3 + 2 + 1 = 10
...
sumTo(100) = 100 + 99 + ... + 2 + 1 = 5050
Сделайте три варианта решения:
С использованием цикла.
Через рекурсию, т.к. sumTo(n) = n + sumTo(n-1) для n > 1.

Код:

      var userAnswer = +prompt('Введите число', 0);

      // Loop
      function sumTo1(n) {
        var sum = 0;
        var i = 0;

        while (i < n) {
          i++;
          sum += i;
        }
        return sum;
      }
      document.write(sumTo1(userAnswer) + ' - Loop' + '<br>');

      // Recursion
      function sumTo2(n) {
        if (n == 1 || n == 0) return n;
        return n + sumTo2(n - 1);
      }
      document.write(sumTo2(userAnswer) + ' - Recursion' + '<br>');
